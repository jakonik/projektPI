 21 /** sha512 function */
 22 const sha512 = function (password, salt) {
 23   let hash = crypto.createHmac('sha512', salt); /** Hashing algorithm sha512 */
 24   hash.update(password);
 25   let value = hash.digest('hex');
 26   return value
 27 };
